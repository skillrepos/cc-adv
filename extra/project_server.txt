#!/usr/bin/env python3
"""Lab 5 (Capstone): Your own MCP server -- "project-health".

An MCP server is just a process that speaks the Model Context Protocol over
stdin/stdout. FastMCP (from the official Python MCP SDK) hides the protocol:
you write plain Python functions, decorate them with @mcp.tool(), and the
docstring + type hints become the tool's documentation and input schema --
exactly what Claude reads when it decides which tool to call.

This server gives Claude Code three "project health" tools for THIS repo:
  run_tests()     -- run the app/ test suite and report what passed/failed
  count_todos()   -- count TODO/FIXME comments across the project's code
  project_stats() -- file and line counts by language

[Lab 5 - Advanced Claude Code - Rev 1.0 - 07/07/26]
"""
import pathlib
import subprocess
import sys

from mcp.server.fastmcp import FastMCP

# The repo root, resolved relative to this file -- so the tools work no matter
# which directory the server process is started from.
ROOT = pathlib.Path(__file__).resolve().parent.parent

mcp = FastMCP("project-health")


@mcp.tool()
def run_tests() -> str:
    """Run the project's behavior test suite (app/test_app.py) and return the
    full PASS/FAIL output plus the exit code. Use this to find out whether the
    to-do API currently meets its contract and exactly which cases fail."""
    proc = subprocess.run(
        [sys.executable, str(ROOT / "app" / "test_app.py")],
        capture_output=True, text=True, timeout=120,
    )
    output = (proc.stdout + proc.stderr).strip()
    return f"exit code: {proc.returncode}\n{output}"


@mcp.tool()
def count_todos() -> str:
    """Count TODO and FIXME comments in the project's Python and JavaScript
    source files. Returns one line per file that has any, plus a total.
    Use this to gauge how much known-unfinished work the codebase carries."""
    skip = {".venv", "node_modules", ".git", "__pycache__", "extra"}
    lines = []
    total = 0
    for path in sorted(ROOT.rglob("*")):
        if path.suffix not in (".py", ".js") or set(path.parts) & skip:
            continue
        try:
            text = path.read_text(errors="replace")
        except OSError:
            continue
        count = sum(1 for line in text.splitlines() if "TODO" in line or "FIXME" in line)
        if count:
            lines.append(f"{path.relative_to(ROOT)}: {count}")
            total += count
    lines.append(f"TOTAL: {total}")
    return "\n".join(lines)


@mcp.tool()
def project_stats() -> str:
    """Summarize the codebase: number of files and total lines per source
    file type (.py, .js, .md, .json, .yml). Use this for a quick sense of
    the project's size and shape."""
    skip = {".venv", "node_modules", ".git", "__pycache__"}
    stats: dict[str, list[int]] = {}
    for path in ROOT.rglob("*"):
        if path.suffix not in (".py", ".js", ".md", ".json", ".yml") or set(path.parts) & skip:
            continue
        try:
            n = len(path.read_text(errors="replace").splitlines())
        except OSError:
            continue
        files, count = stats.get(path.suffix, [0, 0])
        stats[path.suffix] = [files + 1, count + n]
    lines = [f"{ext}: {files} files, {count} lines" for ext, (files, count) in sorted(stats.items())]
    return "\n".join(lines) or "no source files found"


if __name__ == "__main__":
    # stdio is the default transport: Claude Code starts this process and
    # talks to it over stdin/stdout -- the same kind of server you added
    # with `claude mcp add` as a consumer. Now you're the producer.
    mcp.run()
